function ParseFormula(formulaString: string) {
  let len = formulaString.length;
  let state = 'start';
  let searchFrom = 0;
  let i;
  let target: PlainObject[] = [];
  for (i = 0; i < len; i++) {
    let c = formulaString.substring(i, i + 1);
    if (state == 'start') {
      switch (c) {
        case '+':
        case '-':
        case '*':
        case '/':
        case '(':
        case ')':
        case ',':
        case '%':
        case '=':
          target.push(GetSourceElement(c))
          continue;
        case '>':
        case '<':
          if (i + 2 < len && formulaString.substring(i + 1, i + 2) == '=') {
            let v = c + '=';
            target.push(GetSourceElement(v))
            i++;
            continue;
          }
          target.push(GetSourceElement(c))
          continue;
        case '|':
        case '&':
          if (i + 1 < len && formulaString.substring(i + 1, i + 2) == c) {
            target.push(GetSourceElement(c + c))
            ++i;
            continue;
          }
          return notifyWarning('公式格式不正确')
      }

      if (!IsStringValid(c)) {
        continue;
      }

      searchFrom = i;
      if (c == '"') {
        state = "string";
      } else if (!isNaN(toNumber(c)) || c == '.') {
        state = "number";
      } else {
        state = "custom";
      }
      continue;
    }

    if (state == "string") {
      if (c == '"') {
        state = "start";
      } else if (c == '\\') {
        ++i;
      }
    } else if (state == "number") {
      if (isNaN(toNumber(c)) && c != '.') {
        let v = formulaString.substring(searchFrom, i);
        target.push(GetSourceElement('num', v))
        state = "start";
        --i;
      }
    } else {
      if (!IsLetterOrDigit(c) && c != '_' && c != ':') {
        let v = formulaString.substring(searchFrom, i);
        target.push(GetSourceElement(v))
        state = "start";
        --i;
      }
    }
  }

  if (state == "string") {
    return notifyWarning('公式格式不正确');
  }

  if (state == "number") {
    let v = formulaString.substring(searchFrom, len);
    target.push(GetSourceElement('num', v))
  } else if (state == "custom") {
    let v = formulaString.substring(searchFrom, len);
    target.push(GetSourceElement(v))
  }

  targetList.value = target;
}

function IsLetterOrDigit(char: string): boolean {
  if (char.length !== 1) {
    return false
  }

  const regex = /^[a-zA-Z0-9]$/;
  return regex.test(char);
}

function GetSourceElement(name: string, value: string | null = null) {
  for (let item of commonList.value) {
    if (item.id.toString() == name) {
      let tmpItem = DeepClone(item)
      if (value != null) {
        tmpItem.name = value
      }
      return tmpItem;
    }
  }

  for (let item of customFieldsList.value) {
    if (item.id.toString() == name) {
      let tmpItem = DeepClone(item)
      if (value != null) {
        tmpItem.name = value
      }
      return tmpItem;
    }
  }

  // 处理条件字段
  const conditionFields: Record<string, string> = {
    'pick_mode_large': '拣货方式=大单拣货',
    'pick_mode_basket': '拣货方式=逐筐分拣',
    'pick_order_single': '分拣单类型=一单一货',
    'pick_order_multi': '分拣单类型=一单多货',
    'pick_order_same': '分拣单类型=相同货品',
    'pick_order_second': '分拣单类型=二次分拣'
  };

  if (conditionFields[name]) {
    return { id: name, name: conditionFields[name] };
  }

  return { id: name, name: name }
}

  }

  if (formulaString.length == 0) {
    return notifyWarning('公式不能为空')
  }

  let showArray = [];
  showArray.push({ label: "请输入名称", labelWidth: 100, defaultValue: '', type: 'text', maxLength: '50' });
  let resp = await easy.showDialog(
    TextInput,
    { showArray },
    {
      title: '请输入...',
      customStyle: { width: '500px', height: '200px' },
      closeOnPressEscape: true, defaultFooter: true
    }
  );
  let input = resp?.content1 ?? "";

  if (!IsStringValid(input)) {
    return notifyWarning('请输入有效值')
  }
  try {
    await httpPost('/api/o/cfg_user_data_formula_insert', { type: 7, details: formulaString, formulaName: input })
  } catch (e: any) {
    if (e.message.includes('UK_cfg_ui_layout')) {
      Alert('名称已经存在！')
      return
    }
  }
  easy.reloadTransform('@ScorePerformanceFormulaTemplate')
  notifySuccess('添加成功')
}

function OnGetFormulaString() {
  let sb: string[] = [];
  for (let item of targetList.value) {
    if (item.type === 'number') {
      sb.push(item.name.toString())
    } else if (item.id) {
      sb.push(item.id.toString())
    }
  }
  return sb.join('');
}

async function OnChangedTemplateList(newVal: ListId, oldVal: ListId) {
  if (newVal != oldVal) {
    let recId = toNumber(newVal);
    if (recId == 0) return;
    let row = await httpPost('/api/o/cfg_user_data_get', { recId: recId }).then(res => {
      if (res && res.length > 0) return res[0]
      else return null
    });
    if (row != null) {
      ParseFormula(row.data)
    }
  }
}
