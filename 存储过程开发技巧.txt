DECLARE stockout_order_consign_cursor CURSOR FOR
SELECT tmp.stockout_id, tmp.src_order_id, tmp.stockout_no, (tmp.num - SUM(IF(somo.operate_type = 0, IFNULL(somo.num, 0), 0))) AS unfinish
FROM (
	SELECT so.stockout_id, so.src_order_id, so.stockout_no, SUM(sod.num) num
	FROM stockout_order so
	LEFT JOIN stockout_order_detail sod ON sod.stockout_id = so.stockout_id
	INNER JOIN tmp_xchg tx ON tx.f1 = sod.stockout_id
	WHERE so.status = 55 AND sod.is_package = 0 AND tx.f3 = 1 AND so.consign_status & 1 = 0
	GROUP BY so.stockout_id
) tmp
LEFT JOIN stockout_order_multi_operate somo ON somo.stockout_id = tmp.stockout_id
GROUP BY tmp.stockout_id
