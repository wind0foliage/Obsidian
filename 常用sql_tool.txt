SELECT st.package_id, IFNULL(ste.reserve_v3, 0), st.receiver_province, st.receiver_city, st.receiver_district, IFNULL(so.logistics_id, st.logistics_id), st.goods_type_count, st.trade_no, st.warehouse_id, st.owner_id, IFNULL(ss.shop_id, 0), st.weight, st.volume, st.shop_name
FROM sales_trade st 
LEFT JOIN sales_trade_ex ste ON st.trade_id = ste.trade_id
LEFT JOIN sys_shop ss ON ss.shop_name = st.shop_name AND ss.owner_id = st.owner_id AND ss.is_disabled = 0
LEFT JOIN stockout_order so ON so.src_order_id = st.trade_id AND so.src_order_type = 1
WHERE st.trade_id = 54994;
