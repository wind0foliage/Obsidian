			LEFT JOIN stock_spec_detail ssd ON ssd.rec_id = sodp.stock_spec_detail_id
			WHERE sod.stockout_id = P_StockoutId AND sod.is_package < 2;

		IF V_SodpCount > @tmp_stockout_delete_goods_position_limit_num THEN
			INSERT INTO order_latency_sync(order_id, type, operate_type) VALUES(P_StockoutId, 0, 1);
		ELSE
			CALL I_STOCKOUT_DELETE_GOODS_POSITION(P_StockoutId);
		END IF;
	END IF;
END ;
